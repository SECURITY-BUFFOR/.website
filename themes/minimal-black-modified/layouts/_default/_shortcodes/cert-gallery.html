{{- $images := split (.Get "images") "," -}} {{- $issuers := split (.Get "issuers") "," -}} {{- $names := split (.Get "names") "," -}}
{{- range $index, $src := $images -}} {{- $image := trim $src " " -}} {{- $issuer := trim (index $issuers $index) " " -}} {{- $name := trim (index $names $index) " " -}} {{- if $image -}} {{- $label := default $image $name -}} {{- $alt := cond $issuer (printf "%s - %s" $issuer $label) $label -}} {{- end -}} {{- end -}}
